% =============================================================================
% CareerForge CV Template — Two Column (paracol, modern, full-page)
% Class:   cfcv.cls
% Compile: lualatex main_example_two_col.tex    (run from cv/ directory)
% Output:  main_example_two_col.pdf             (target: 1 page, full-bleed)
%
% Layout:  Left sidebar 32%  |  thin accent rule  |  Right main 68%
%          White background; FontAwesome 4 icons via cfcv.cls symbols.
%
% Header alignment trick:
%   Left column starts empty. \switchcolumn* switches to right, which gets
%   the full header (name + tagline + contact + rule). The second \switchcolumn*
%   advances the left column to match the right header height automatically.
%   Result: About Me and Work Experience start at exactly the same Y position
%   with no hardcoded \vspace.
%
% Page fill strategy:
%   Left column  — 2 × \vfill distributes breathing room between sections.
%   Right column — 1 × \vfill before Publications pins it to the page bottom.
%
% ATS: skill names are real text; bars are graphic-only.
%      paracol outputs columns sequentially. No letter-spacing anywhere.
%
% Tokens:  Fields marked  % {{TOKEN}}  are replaced by the AI /apply skill.
% =============================================================================

\documentclass[10pt, a4paper]{cfcv}

% ── Override cfcv.cls skip amounts (tighter rhythm for the right column) ───────
\setlength{\bigskipamount}{3pt}
\setlength{\medskipamount}{3pt}
\setlength{\smallskipamount}{2pt}

% ── Paracol: 32 / 68 split ─────────────────────────────────────────────────────
\columnratio{0.32}
\setlength{\columnsep}{0.65cm}

% ── Thin accent rule at column boundary ────────────────────────────────────────
\AddToShipoutPictureBG{%
  \begin{tikzpicture}[remember picture,overlay]
    \draw[accent!22, line width=0.4pt]
      ([xshift=0.350\paperwidth]current page.north west)
      --
      ([xshift=0.350\paperwidth]current page.south west);
  \end{tikzpicture}}

% ── Sidebar section header ─────────────────────────────────────────────────────
\newcommand{\sidesection}[1]{%
  \vspace{0.40cm}%
  {\color{accent}\small\bfseries\MakeUppercase{#1}}\\[-1ex]%
  {\color{accent}\rule{\linewidth}{1.2pt}\par}\smallskip}

% ── Icon-prefixed contact item ─────────────────────────────────────────────────
\newcommand{\iconitem}[2]{%
  {\color{accent}#1}\hspace{0.35em}{\footnotesize\color{body}#2}\par\vspace{0.07cm}}

% ── Skill progress bar ─────────────────────────────────────────────────────────
\newcommand{\skillbar}[2]{%
  {\footnotesize\color{body}\textbf{#1}}\\[0.02em]%
  \begin{tikzpicture}
    \fill[black!10] (0,0) rectangle (\linewidth,3.5pt);
    \fill[accent]   (0,0) rectangle (#2\linewidth,3.5pt);
  \end{tikzpicture}\par\vspace{0.13cm}}

% ── Two-sub-column entry ───────────────────────────────────────────────────────
% #1=org  #2=location  #3=dates (empty → omitted)
% #4=role/degree/title  #5=description (empty → omitted)
\newcommand{\jobentry}[5]{%
  \smallskip\noindent
  \begin{minipage}[t]{0.33\linewidth}
    {\footnotesize\bfseries\color{body}\MakeUppercase{#1}\par}%
    {\scriptsize\color{LightGrey}#2\par}%
    \ifx&#3&\else{\scriptsize\color{accent}#3\par}\fi%
  \end{minipage}%
  \hspace{0.01\linewidth}%
  \begin{minipage}[t]{0.66\linewidth}
    {\footnotesize\bfseries\color{body}#4\par}%
    \ifx&#5&\else{%
      \scriptsize\linespread{1.35}\selectfont\color{body}#5\par}\fi%
  \end{minipage}%
  \par\vspace{0.14cm}}
